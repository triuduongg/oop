void checkLowStockAlert() {
    // Dùng Stream API để lọc danh sách, chỉ lấy các sản phẩm có số lượng chạm mức tối thiểu (isLow)
    List<Product> lowProducts = products.stream().filter(Product::isLow).collect(Collectors.toList());
    // Kiểm tra nếu danh sách sản phẩm sắp hết hàng không bị trống
    if (!lowProducts.isEmpty()) {
        // Tạo đối tượng StringBuilder để nối chuỗi tối ưu hiệu suất, chứa tiêu đề cảnh báo
        StringBuilder sb = new StringBuilder("⚠️ CẢNH BÁO: CÁC SẢN PHẨM CẦN NHẬP THÊM HÀNG\n\n");
        // Vòng lặp duyệt qua từng sản phẩm trong danh sách sắp hết
        for (Product p : lowProducts) {
            // Gắn tên, mã SP, số lượng tồn hiện tại và mức tối thiểu của từng sản phẩm vào chuỗi
            sb.append("- ").append(p.name).append(" (Mã: ").append(p.id).append(")").append(" | Tồn hiện tại: ").append(p.quantity).append(" | Tối thiểu: ").append(p.minStock).append("\n");
        }
        // Thêm câu nhắc nhở ở cuối hộp thoại
        sb.append("\nVui lòng kiểm tra và lên kế hoạch nhập kho!");
        // Hiển thị hộp thoại (Popup) cảnh báo với thông điệp đã tạo, kèm icon Warning (Tam giác vàng)
        JOptionPane.showMessageDialog(this, sb.toString(), "Cảnh báo Tồn kho", JOptionPane.WARNING_MESSAGE);
    }
}
void buildUI() {

        // Thêm trang Lịch sử giao dịch vào contentArea
        contentArea.add(buildTransPanel(), "trans");
        // Thêm trang Thống kê vào contentArea
        contentArea.add(buildStatsPanel(), "stats");
        // Lệnh cho CardLayout hiển thị màn hình "dashboard" lên đầu tiên
        cards.show(contentArea, "dashboard");
        // Gọi hàm làm mới dữ liệu cho Dashboard ngay khi vừa nạp
        refreshDashboard();
    } else { // Nếu không phải quản lý kho (tức là Quản lý nhân sự)

        // Hiển thị màn hình quản lý tài khoản lên đầu tiên
        cards.show(contentArea, "accounts");

    // Gắn thanh Top Bar (Header) lên phía Bắc (Trên cùng) của cửa sổ
    add(buildTopBar(), BorderLayout.NORTH);

    add(buildSidebar(), BorderLayout.WEST);

    add(contentArea, BorderLayout.CENTER);

JPanel buildSidebar() {

    // Căn lề trái cho nhãn
    lbl.setAlignmentX(Component.LEFT_ALIGNMENT);

    boolean first = true;

    for (String[] item : items) {

JToggleButton buildNavBtn(String label, String card) {
    // Khởi tạo JToggleButton và ghi đè sự kiện vẽ để làm giao diện menu cực phẳng
    JToggleButton btn = new JToggleButton() {
        @Override protected void paintComponent(Graphics g) {
            // Ép sang Graphics2D
            Graphics2D g2 = (Graphics2D) g.create();
            // Khử răng cưa
            g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
            // Lấy trạng thái xem nút có đang được chọn hay không
            boolean sel   = isSelected();
            // Lấy trạng thái xem chuột có đang lướt qua (hover) hay không
            boolean hover = getModel().isRollover();
            // Nếu đang được chọn
            if (sel) {
                // Đổ nền nút màu xanh dương sáng
                g2.setColor(C_BLUE);
                // Vẽ hình nền bo góc
                g2.fillRoundRect(8, 2, getWidth()-16, getHeight()-4, 8, 8);
            } else if (hover) { // Nếu chỉ hover chuột qua
                // Đổ nền màu xanh xám nhẹ
                g2.setColor(C_SIDEBAR_HL);
                // Vẽ hình nền bo góc
                g2.fillRoundRect(8, 2, getWidth()-16, getHeight()-4, 8, 8);
            }
            // Tính tọa độ Y giữa tâm nút
            int cy = getHeight()/2;
            // Chọn màu cho icon chấm tròn nhỏ: Trắng nếu đang chọn, Xám nếu không
            g2.setColor(sel ? C_WHITE : new Color(100,116,139));
            // Vẽ icon chấm tròn đại diện cho menu
            g2.fillOval(20, cy-4, 8, 8);
            // Chữ sẽ in đậm nếu đang chọn, ngược lại in thường
            g2.setFont(sel ? new Font("Segoe UI", Font.BOLD, 13) : F_BODY);
            // Chọn màu chữ
            g2.setColor(sel ? C_WHITE : new Color(203,213,225));
            // Vẽ chữ của menu lên nút
            g2.drawString(label, 38, cy+5);
            // Hủy đối tượng vẽ
            g2.dispose();
        }
    };
    // Khóa cứng kích thước nút là 210x42 pixel
    btn.setPreferredSize(new Dimension(210, 42));
    btn.setMaximumSize(new Dimension(210, 42));
    // Tắt hiển thị viền và nền mặc định của Java Swing
    btn.setOpaque(false); btn.setContentAreaFilled(false);
    btn.setBorderPainted(false); btn.setFocusPainted(false);
    // Bật con trỏ hình bàn tay
    btn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
    // Căn trái
    btn.setAlignmentX(Component.LEFT_ALIGNMENT);
    // Thêm sự kiện khi nút được Click
    btn.addItemListener(e -> {
        // Chỉ xử lý khi trạng thái chuyển sang SELECTED (Tránh click nhiều lần)
        if (e.getStateChange() == ItemEvent.SELECTED) {
            // Nếu là trang Thống kê Kho, gọi hàm nạp lại dữ liệu thống kê Kho
            if ("stats".equals(card))     refreshStats();
            // Nếu là trang Thống kê Nhân sự, nạp lại biểu đồ nhân sự
            if ("stats_acc".equals(card)) refreshAccStats();
            // Nếu là trang Giao dịch, nạp lại bảng Lịch sử GD
            if ("trans".equals(card))     refreshTransTable();
            // Nếu là Tổng quan, tính toán lại các Card Tổng
            if ("dashboard".equals(card)) refreshDashboard();
            // Nếu vào trang Nhập kho, dựng lại panel Nhập để reset ô chọn
            if ("import".equals(card))    rebuildImport();
            // Nếu vào trang Xuất kho, dựng lại panel Xuất
            if ("export".equals(card))    rebuildExport();
            // Lệnh CardLayout hiển thị đúng tên trang (card) tương ứng
            cards.show(contentArea, card);
            // Bắn log xuống thanh Status Bar dưới đáy màn hình
            setStatus("Đang xem: " + label);
        }
    });
    // Trả về nút đã hoàn thiện
    return btn;
}

void refreshDashboard() {
    // Nếu biến chưa khởi tạo thì thoát để tránh lỗi NullPointer
    if (statTotal == null) return;
    // Đếm tổng số lượng Sản phẩm có cờ isLow() = true
    long low = products.stream().filter(Product::isLow).count();
    // Đếm tổng số lượng SP chạm mốc 0
    long out = products.stream().filter(Product::isEmpty).count();
    // Vòng lặp mapToDouble để tính tổng tiền (Giá * Số lượng tồn) của toàn bộ kho
    double val = products.stream().mapToDouble(p -> p.price * p.quantity).sum();
    // Update Text: Tổng số mã sản phẩm trong kho
    statTotal.setText(String.valueOf(products.size()));
    // Update Text: Định dạng số tổng tiền hiển thị hàng nghìn có dấu phẩy (%,.0f)
    statValue.setText(String.format("%,.0f", val));
    // Update Text: Số lượng sắp hết
    statLow.setText(String.valueOf(low));
    // Update Text: Số lượng đã hết
    statOut.setText(String.valueOf(out));
    // Làm mới lại dữ liệu hiển thị trên Bảng danh sách
    if (tblDashModel != null) {
        // Xóa sạch dữ liệu cũ
        tblDashModel.setRowCount(0);
        // Quét lại mảng Products và nạp từng dòng
        for (Product p : products) {
            // Nạp 1 Row (hàng) dữ liệu vào Model
            tblDashModel.addRow(new Object[]{
                p.id, p.name, p.category, p.brand, String.format("%,.0f đ", p.price), p.quantity, p.statusText()
            });
        }
    }
}

void exportTransToCSV() {
    // Chặn không cho xuất nếu bảng rỗng
    if (tblTransModel.getRowCount() == 0) { warn("Không có dữ liệu để xuất!"); return; }
    // Bật hộp thoại File Dialog chuẩn của Hệ điều hành để chọn chỗ Save
    JFileChooser fileChooser = new JFileChooser();
    fileChooser.setDialogTitle("Chọn nơi lưu báo cáo giao dịch");
    // Gợi ý tên file tự động có kèm Ngày Tháng Năm (ví dụ: LichSuGiaoDich_25042026.csv)
    fileChooser.setSelectedFile(new File("LichSuGiaoDich_" + new SimpleDateFormat("ddMMyyyy").format(new Date()) + ".csv"));
    
    // Nếu người dùng ấn Save trên hộp thoại
    if (fileChooser.showSaveDialog(this) == JFileChooser.APPROVE_OPTION) {
        // Dùng khối try-with-resources để tự động đóng file (Tránh lỗi thất thoát bộ nhớ). Format chuẩn UTF-8
        try (PrintWriter pw = new PrintWriter(new OutputStreamWriter(new FileOutputStream(fileChooser.getSelectedFile()), "UTF-8"))) {
            // RẤT QUAN TRỌNG: Chèn ký tự BOM (Byte Order Mark) để phần mềm Microsoft Excel nhận diện được font tiếng Việt UTF-8
            pw.write('\ufeff'); 
            // Dòng 1: In Tiêu đề Header của file CSV (phân cách bằng dấu phẩy)
            pw.println("Mã GD,Mã SP,Tên sản phẩm,Loại,Số lượng,Người thực hiện,Ngày giờ,Ghi chú");
            
            // Duyệt từng hàng đang được hiển thị trên Bảng giao diện
            for (int r = 0; r < tblTransModel.getRowCount(); r++) {
                // Khởi tạo StringBuilder để nối dòng
                StringBuilder row = new StringBuilder();
                // Duyệt qua từng Cột của 1 Hàng
                for (int c = 0; c < tblTransModel.getColumnCount(); c++) {
                    // Lấy giá trị chữ tại ô (r,c)
                    String val = tblTransModel.getValueAt(r, c) != null ? tblTransModel.getValueAt(r, c).toString() : "";
                    // Ép nội dung vào trong nháy kép ("nội dung") để tránh bị lỗi nếu Ghi chú của ng dùng có chứa dấu phẩy
                    row.append("\"").append(val.replace("\"", "\"\"")).append("\"");
                    // Chèn dấu phẩy phân cách sang ô tiếp theo
                    if (c < tblTransModel.getColumnCount() - 1) row.append(",");
                }
                // Ghi 1 dòng rành mạch ra file
                pw.println(row.toString());
            }
            // Thông báo thành công!
            JOptionPane.showMessageDialog(this, "Xuất báo cáo thành công!", "Hoàn tất", JOptionPane.INFORMATION_MESSAGE);
        } catch (Exception ex) {
            // Ném Popup lỗi nếu đường dẫn hỏng hoặc đầy ổ cứng
            warn("Lỗi khi xuất file: " + ex.getMessage());
        }
    }
}

        @Override protected void paintComponent(Graphics gx) {
            super.paintComponent(gx);
            Graphics2D g2 = (Graphics2D) gx.create();
            // Khử răng cưa mịn ảnh
            g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);

            // Bắt đầu nhặt dữ liệu tổng tiền Group By theo Category
            Map<String, Double> vals = new HashMap<>();
            double max = 0; // Trục Y lớn nhất để quy chiếu thang đo tỷ lệ
            for (Product p : products) {
                double v = p.price * p.quantity;
                // Nếu danh mục đã có trong Map thì cộng dồn, nếu chưa thì bắt đầu với 0.0 + v
                vals.put(p.category, vals.getOrDefault(p.category, 0.0) + v);
            }
            // Tìm Cột có số tiền lớn nhất để làm Cột MAX cao đụng trần nhà
            for (double v : vals.values()) if (v > max) max = v;

            // Nếu kho rỗng 0đ -> Hiện chữ nhắc nhở "Chưa có dữ liệu"
            if (max == 0 || vals.isEmpty()) {
                g2.setColor(C_TEXT3); g2.drawString("Chưa có dữ liệu thống kê", getWidth()/2 - 60, getHeight()/2); return;

            int graphW = w - 2*padding, graphH = h - 2*padding; // Chiều rộng/cao thật dùng để vẽ

            g2.setColor(C_BORDER); g2.drawLine(padding, h-padding, w-padding, h-padding);

            // Toán học: Tính kích thước chiều rộng từng cột dựa vào số lượng Cột danh mục tổng
            int numBars = vals.size();
            int barWidth = Math.min(60, graphW / numBars / 2); // Kích thước Cột tối đa 60px
            // Tính khoảng cách lề xen kẽ giữa các Cột
            int spacing = (graphW - (barWidth * numBars)) / (numBars + 1);
            int x = padding + spacing; // Tọa độ X (cột đầu tiên)
            
            // Bảng màu gradient (màu thay đổi qua từng vòng lặp)
            Color[] colors = {C_BLUE, C_GREEN, C_AMBER, new Color(139, 92, 246), new Color(236, 72, 153)};
            int cIdx = 0;

            // Duyệt Hashmap vẽ các cột
            for (Map.Entry<String, Double> e : vals.entrySet()) {
                // Tính chiều cao cột theo tỷ lệ của MAX (v/max * 100%)
                int barH = (int) ((e.getValue() / max) * graphH);
                // Vì tọa độ Java (0,0) nằm ở Top-Left nên tọa độ Y phải đảo ngược lật từ dưới lên
                int y = h - padding - barH;

                // Chọn màu tương ứng, đổ viền bo góc cột (6px bo)
                g2.setColor(colors[cIdx % colors.length]); g2.fillRoundRect(x, y, barWidth, barH, 6, 6);

                // Dựng Text dán nhãn của Danh mục dưới chân Cột (Label dưới trục X)
                g2.setColor(C_TEXT2); g2.setFont(F_SMALL);
                String label = e.getKey();
                int lw = g2.getFontMetrics().stringWidth(label); // Lấy độ rộng string để Căn giữa text theo Cột
                g2.drawString(label, x + (barWidth - lw)/2, h - padding + 15);
                
                // Nhích hệ tọa độ X dịch sang phải để chuẩn bị vẽ vòng lặp cột tiếp theo
                x += barWidth + spacing; cIdx++;
            }
            g2.dispose();
        }

    JPanel centerSplit = new JPanel(new BorderLayout(0, 16));
    centerSplit.setBackground(C_BG);
    centerSplit.add(grid, BorderLayout.NORTH); centerSplit.add(chartCard, BorderLayout.CENTER);
    
    // Gắn thẻ ClientProperty tĩnh lên Panel Tổng để các sự kiện bên ngoài gọi vào lấy dc Model Bảng
    outer.putClientProperty("mCat", mCat);
    outer.putClientProperty("mLow", mLow);
    outer.add(centerSplit, BorderLayout.CENTER);

    // Load Data
    refreshStatsPanel(outer);
    return outer;
}

void refreshStats() {
    // Vòng quét đệ quy qua các giao diện để bắt được Form "Thống Kê" (Chứa mCat) sau đó ép Load lại dữ liệu
    for (Component c : contentArea.getComponents()) {
        if (c instanceof JPanel) {
            JPanel p = (JPanel) c;
            if (p.getClientProperty("mCat") instanceof DefaultTableModel) {
                refreshStatsPanel(p);
            }
        }
    }
}

void refreshStatsPanel(JPanel panel) {
    // Nối với dữ liệu Bảng ngoài ClientProperty
    DefaultTableModel mCat = (DefaultTableModel) panel.getClientProperty("mCat");
    DefaultTableModel mLow = (DefaultTableModel) panel.getClientProperty("mLow");
    if (mCat == null || mLow == null) return;
    
    // Reset Data mCat
    mCat.setRowCount(0);
    // Nhóm sản phẩm (GroupBy) bằng Java 8 Stream
    products.stream().collect(Collectors.groupingBy(p -> p.category))
        .forEach((cat, list) -> {
            // Cộng tổng Số lượng và Giá trị Tồn
            int tot    = list.stream().mapToInt(p -> p.quantity).sum();
            double val = list.stream().mapToDouble(p -> p.price * p.quantity).sum();
            mCat.addRow(new Object[]{cat, list.size(), tot, String.format("%,.0f", val)});
        });
        
    // Reset Data mLow (Sắp hết hàng)
    mLow.setRowCount(0);
    // Quét stream nhặt Data isLow (Cảnh báo Tồn)
    products.stream().filter(Product::isLow).forEach(p -> mLow.addRow(new Object[]{p.id, p.name, p.quantity, p.minStock}));
        
    // Lệnh yêu cầu Hệ thống vẽ lại Biểu đồ Cột ngay lập tức để đồng bộ real-time
    if (chartView != null) chartView.repaint(); 
}

// N5: Định dạng Data Bảng JTable trở nên phẳng, chuyên nghiệp
JTable buildTable(DefaultTableModel model) {
    JTable t = new JTable(model);
    t.setBackground(C_WHITE); t.setForeground(C_TEXT);
    // Font hiển thị thông thường, độ cao Row 36px cho chạm ngón tay

    t.setShowGrid(false); t.setIntercellSpacing(new Dimension(0,0));

    t.setFillsViewportHeight(true);

    t.getTableHeader().setForeground(C_TEXT2);

    t.getTableHeader().setPreferredSize(new Dimension(0,38));
    t.getTableHeader().setBorder(BorderFactory.createMatteBorder(0,0,1,0,C_BORDER));
    // Bộ máy Render (In chữ lên Cell)
    t.setDefaultRenderer(Object.class, new DefaultTableCellRenderer() {
        @Override public Component getTableCellRendererComponent(JTable tbl, Object v, boolean sel, boolean foc, int r, int c) {
            super.getTableCellRendererComponent(tbl, v, sel, foc, r, c);
            // Margin thụt lề 12px
            setBorder(new EmptyBorder(0,12,0,12));
            setFont(F_BODY);
            // Chức năng dòng Zebra chẵn/lẻ đổi màu xám trắng
            setBackground(sel ? C_SEL : (r%2==0 ? C_WHITE : C_ROW_ALT));
            setForeground(C_TEXT);
            // Tính năng nổi bật Cảnh báo Chữ theo Nội dung chuỗi Trạng Thái
            if (v != null) {
                switch (v.toString()) {
                    case "Hết hàng": setForeground(C_RED);   setFont(F_LABEL); break; // Hết thì tô Đỏ
                    case "Sắp hết":  setForeground(C_AMBER); setFont(F_LABEL); break; // Thiếu thì tô Vàng

            }
            return this;
        }
    });
    // Gắn TableRowSorter cho phép Bấm tiêu đề Cột -> Tự Sắp xếp list (A-Z)
    t.setRowSorter(new TableRowSorter<>(model));
    return t;

// N5: Định dạng các Ô Search / Nhập Text có bóng mờ (Placeholder hint)
JTextField buildTextField(String placeholder, int w) {
    JTextField f = new JTextField();
    f.setFont(F_BODY); f.setBackground(C_WHITE);
    f.setForeground(C_TEXT); f.setCaretColor(C_BLUE); // Con trỏ nháy màu xanh dương
    // Kẻ viền xám 1px
    f.setBorder(BorderFactory.createCompoundBorder(BorderFactory.createLineBorder(C_BORDER, 1), new EmptyBorder(6,10,6,10)));
    f.setPreferredSize(new Dimension(w > 0 ? w : 300, 34)); // Chiều rộng cài tĩnh bằng tham số W
    // Nhúng giả lập Hint Placeholder nếu có
    if (!placeholder.isEmpty()) {
        f.setText(placeholder); f.setForeground(C_TEXT3); // Text xám siêu mờ
        f.addFocusListener(new FocusAdapter() {
            @Override public void focusGained(FocusEvent e) { // Bấm chuột vào xóa Hint
                if (f.getText().equals(placeholder)) { f.setText(""); f.setForeground(C_TEXT); }
            }
            @Override public void focusLost(FocusEvent e) { // Bỏ chuột ra nếu trống -> Nạp lại Hint
                if (f.getText().isEmpty()) { f.setText(placeholder); f.setForeground(C_TEXT3); }
            }
        });
    }
    return f;
}

static JButton createBtn(String text, Color bg, Color fg) {
    JButton btn = new JButton(text) {
        @Override protected void paintComponent(Graphics g) {
            Graphics2D g2 = (Graphics2D) g.create();
            g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
            // Chỉnh sắc độ Nút Đậm khi ấn xuống, Nhạt đi khi lướt lên
            Color c = getModel().isPressed()  ? bg.darker() : getModel().isRollover() ? bg.brighter() : bg;
            g2.setColor(c); g2.fillRoundRect(0,0,getWidth(),getHeight(),8,8); // Tô góc nén 8px
            g2.setColor(C_BORDER); g2.setStroke(new BasicStroke(1f));
            g2.drawRoundRect(0,0,getWidth()-1,getHeight()-1,8,8);
            g2.dispose(); super.paintComponent(g);
        }
    };
    btn.setFont(new Font("Segoe UI", Font.BOLD, 13));
    btn.setForeground(fg);
    btn.setOpaque(false); btn.setContentAreaFilled(false);
    btn.setBorderPainted(false); btn.setFocusPainted(false); // Xóa chấm khung đứt
    btn.setBorder(new EmptyBorder(7,16,7,16)); // Đệm
    btn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR)); // Đổi Trỏ tay
    return btn;
}
